/*****************************************************************
Function Name :Support Type Hiding
Name : Nandhini Vurimella
Description : For hiding support type
*****************************************************************/
mPN=selectedProductNumberInt_avg;
currdate=getdate();
supporttype="";
ret=false;
results=bmql("select Existing_Suprt_Type,Start_Date,End_Date from BRCM_Part_Master_Prc where MPN=$mPN ORDER BY Start_Date ASC");
for res in results
{ //viper-870; Added condition for Start Date and end date check condition
 if(strtojavadate(get(res,"Start_Date"),"yyyy-MM-dd HH:mm:ss") <= currdate AND strtojavadate(get(res,"End_Date"),"yyyy-MM-dd HH:mm:ss") >= currdate)
 {
 supporttype=get(res,"Existing_Suprt_Type");
 }
}

if(supporttype=="" OR isnull(supporttype)==true)
{
 ret=true;
}
else
{
 ret=false;
}


return ret;
