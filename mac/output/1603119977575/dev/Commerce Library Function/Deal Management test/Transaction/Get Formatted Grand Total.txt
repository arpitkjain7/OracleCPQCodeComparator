/************************************************************************************************************
Description: Function to  return grand total in #####.## format since oracle is rounding off to #####.# when decimal value is zero.
Return type: string
History:     Date            Author             Comment
      03/JULY/2019    Raghavendra Kumar    Created for resolving BROCAD2-3058
*************************************************************************************************************/
resultString = inputGrandTotalString;
numberPart = "";
grandTotalArray = split(inputGrandTotalString,".");
sb = stringbuilder("");
if(sizeofarray(grandTotalArray) == 2)
{
 numberPart = grandTotalArray[0];
 decimalPart = grandTotalArray[1];
 if(len(decimalPart) == 1)
 {
  decimalPart = decimalPart+"0";
  sb = stringbuilder(numberPart,".",decimalPart);
  resultString = sbtostring(sb);
 }
}
return resultString;
