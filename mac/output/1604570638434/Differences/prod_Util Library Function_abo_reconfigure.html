t1//Function Name: abo_reconfiguret
2//    input: lineBomId(String), lineTrx_Date(String), bs_Id(String), line_Num(String). 
3//    output:resultMap
4// Description:
5//    This is the entry BML for ABO reconfig operation.This BML captures logic specific to ABO reconfig flow, 
6//    This is invoked from the reconfig action of both main doc and sub doc.
7// Details:
8//     This will calculate PAC BOM and config BOM and set them in user session map to be picked up by configurator.
9 
10abocontext = util.abo_getContext();
11bDiagnosisOff=jsonget(abocontext, "AboDiagnosticDisabled", "boolean", true);  
12callContext="reconfig";
13CREATED="CREATED";
14MODIFY = "MODIFY";
15 
16if(not bDiagnosisOff){//log input
17  inputJson=json();
18  jsonput(inputJson,"assetkey", lineBomId);
19  jsonput(inputJson,"lineTxnDate", lineTrx_date);
20  jsonput(inputJson, "bs_id", bs_id);
21  jsonput(inputJson, "line_num", line_Num);
22  dummy=util.abo_addDiagnosticInfo(callContext+"_input", inputJson,"addToSummary");
23}
24 
25//step1  collect input and add default vaule to make it easier to process later
26lineNum = atoi(line_Num);
27lineTrxDate =  getDate(false);
28if(lineTrx_Date<>""){
29    lineTrxDate = strtojavadate(lineTrx_Date, "MM/dd/yyyy");
30}
31strTransDate = datetostr(lineTrxDate, "yyyy-MM-dd HH:mm:ss"); 
32 
33//read the value of current process info from the aboContext 
34curProcessInfo=jsonget(abocontext, "curProcessInfo", "json", json());
35isExternal=jsonget(curProcessInfo, "isExternal", "string");
36curProcessName=jsonget(curProcessInfo, "commerceProcessName", "string");
37 
38//step2  retrieve fulfillment/action info & BOM from the reconfig line to allow early validation
39currentQuoteBom = json();
40fulfillmentStatus = "";
41actionCodeConfigBom = "";
42site_url =  jsonpathgetsingle(abocontext, "$.siteInfo.siteUrl");
43 
44if(isExternal == "true"){// if it is an external process, retrieve the BOM from the config integration table
45   currentQuoteBom = getconfigbom(lineNum,true /*flatten*/ );   
46   
47   //fulfillment status is not available in bom, so we will retrieve from configBomInstannce
48   configUrl = site_url + "/rest/v3/configBomInstance/" + line_Num; 
49   configPayload =json();
50   responseConfigBom =  util.abo_invokeRESTService("Get",configUrl,configPayload);
51   fulfillmentStatus = jsonpathgetsingle(responseConfigBom , "$.fulfillmentStatus.lookupCode","String");  
52   actionCodeConfigBom = jsonget(responseConfigBom , "actionCode", "String", ""); 
53   
54} else {//internal process, retrieve bom from cpq commerce via getbom api
55   lineField =  string[3];
56   lineField[0]="oRCL_ABO_ActionCode_l";
57   lineField[1] = "itemInstanceName_l";
58   lineField[2] = "requestDate_l";
59   lineField[3] = "fulfillmentStatus_l";   
60   currentQuoteBom =   getbom(atoi(bs_Id), lineNum, lineField, 
61                                                  false,  //validate
62                                                  true,  //flatten
63                                                  true); //isSalesBom
64              
65   //retrieve  fulfillmentStatus for internal orders
66   fieldJson = jsonget(currentQuoteBom, "fields","json", json());
67   fulfillmentStatus = jsonget(fieldJson, "fulfillmentStatus_l","String");
68   actionCodeConfigBom = jsonget(fieldJson, "oRCL_ABO_ActionCode_l","string", "");
69}
70 
71//step3  validate to see whether current line is still allowed to reconfig   
72if ( (fulfillmentStatus <> CREATED) AND (isnull(fulfillmentStatus)==false))  {
73   throwerror("This configuration is no longer eligible for reconfiguration since it is in being fulfilled/fulfilled state.");
74}
75 
76curActionInfo = jsonpathgetsingle(abocontext, "$.ALL_ACTIONS."+actionCodeConfigBom, "json", json());
77curActionCategory = jsonget(curActionInfo, "CATEGORY", "string", MODIFY);
78if(curActionCategory <> MODIFY){ 
79  actionDisplayName = jsonget(curActionInfo, "DISPLAY", "string", actionCodeConfigBom); 
80  throwerror("This configuration with action of "+actionDisplayName+" cannot be reconfigured."); 
81}
82 
83//step4  getting ProjectedAssetState(without current line ) and projected Config BOM(with current line) 
84pacBomflat = util.abo_generatePAC(lineBomId, lineTrxDate, bs_Id, lineNum, "RECONFIG");
85if(not bDiagnosisOff){
86  dummy=util.abo_addDiagnosticInfo(callContext+"_afterGenPacBom", pacBomflat,"");
87}   
88configBomFlat = jsoncopy(pacBomflat);
89arrToApply =  jsonarray();
90jsonarrayappend(arrToApply, currentQuoteBom);
91 
92configBomFlat =util.abo_apply(configBomFlat, arrToApply);
93 
94 
95//step 5, now prepare the configBom to be in the format configurator expects
96configBom = convertbomtohier(configBomFlat);
97 
98//Include instruction to preseve inputBom for config launch
99jsonput(configBom, "instruction", "preserveInputBom");
100 
101//step 6 place the configbom and pacBom into session cache to be looked up by configurator,
102usersessionremove ("bm_abo_pac_root_json");
103usersessionremove (BM_SALES_ROOT_BOM_ITEM);
104 
105//Store pacBom in global map
106usersessionset("bm_abo_pac_root_json", pacBomflat);
107//set configBom in session scope from where ConfigUI reads BOM
108usersessionset(BM_SALES_ROOT_BOM_ITEM,configBom);
109 
110//lookup the model information for the part number
111resultMap = dict("string");
112partNumber =  jsonget(configBom, "partNumber", "string");
113resultMap = util.abo_getProductModelInfo(partNumber);
114put(resultMap,"bomkey",lineBomId);
115 
116if(not bDiagnosisOff){
117   dummy=util.abo_addDiagnosticInfo(callContext+"_output_pacBom", pacBomflat,"addToSummary");
118   dummy=util.abo_addDiagnosticInfo(callContext+"_output_configBom", configBom,"addToSummary");
119}
120 
121return resultMap;
