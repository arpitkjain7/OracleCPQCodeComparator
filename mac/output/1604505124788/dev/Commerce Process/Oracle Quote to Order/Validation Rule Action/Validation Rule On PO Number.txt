/********************************************************************
Function Name: Validation Rule On PO Number
Description: This rule is created to check if the PO NUMBER contains CPQ ,Contract , N/A
Author:   Nandhini Vurimella
History:     Date          Author                    Comment 
      05-04-2020    Nandhini Vurimella            Initial Version
********************************************************************/
returnmsg=dict("dict<string>");
PartError=dict("string");
errorString="";
PONumberText = purchaseOrderNumber_t;

if( find(PONumberText,"CPQ") <>-1 OR find(PONumberText,"Contract")<>-1 OR find(PONumberText,"N/A") <>-1)
{
   errorString = "PO Number cannot contain CPQ ,Contract ,N/A";
}

put(PartError, "message", errorString);
put(PartError, "location", "Top");
put(returnmsg, "purchaseOrderNumber_t", PartError);
return returnmsg;
