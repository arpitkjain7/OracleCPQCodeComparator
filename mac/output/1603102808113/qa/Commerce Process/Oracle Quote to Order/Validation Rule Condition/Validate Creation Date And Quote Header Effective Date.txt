dateFormat = "MM/dd/yyyy HH:mm:ss";
timeZone = "America/Chicago";
timeResetString = " 00:00:00";

if(find(createdDate_t,"-") > 0)
{
 dateFormat = "yyyy-MM-dd HH:mm:ss";
}
if(createdDate_t <> "" and quoteEffectiveDate_t <> "")
{
 createdDateStr = substring(createdDate_t,0,10) + timeResetString;
 effectiveDateStr = substring(quoteEffectiveDate_t,0,10) + timeResetString;
 createdDateObject = strtojavadate(createdDateStr,dateFormat,timeZone);
 effectiveDateObject = strtojavadate(effectiveDateStr,dateFormat,timeZone);
 if(comparedates(effectiveDateObject,createdDateObject) == -1)
 {
  return true;
 }
}
return false;
