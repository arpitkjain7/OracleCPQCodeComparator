Function Name : Recc_LicenseType_ca
Name : Rohit Ranjan
Description : This rule is created to fetch the value of contract start date from commerce.
*****************************************************************/ 
ret = getdate();
resultSet = bmql("SELECT contractEndDate_t from commerce.transaction");
for each in resultSet{
ret = strtojavadate(get(each,"contractEndDate_t"), "dd/MM/yyyy", "America/Chicago");
}
return ret;
