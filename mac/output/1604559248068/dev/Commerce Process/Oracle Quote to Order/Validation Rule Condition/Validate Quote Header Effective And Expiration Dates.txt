dateFormat = "MM/dd/yyyy HH:mm:ss";
timeZone = "America/Chicago";

if(quoteExpirationDate_t <> "" and quoteEffectiveDate_t <> "")
{
 if(find(quoteEffectiveDate_t,"-") > 0)
 {
  dateFormat = "yyyy-MM-dd HH:mm:ss";
 } 
 effectiveDateObject = strtojavadate(quoteEffectiveDate_t,dateFormat,timeZone);
 expirationDateObject = strtojavadate(quoteExpirationDate_t,dateFormat,timeZone);
 if(comparedates(expirationDateObject,effectiveDateObject) == -1)
 {
  return true;
 }
}
return false;
