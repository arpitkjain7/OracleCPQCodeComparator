t1//Function Name: abo_sortDeltaBomt
2//input: deltaBom(json)
3//output: delta bom with sorted Child array(jsonarray)
4// Description:
5//   This utility BML is used to sort Update Asset's deltaBom so that parents appear before children for non-delete lines, 
6//        and children appear before parents for delete lines.
7//  we do this for 3 reasons:
8// 1. for tree insert, parent has to be inserted before child otherwise we wont'
9//      be able to resolve FK reference correctly.
10//  2. for tree delete, child has to be deleted before parent otherwise it will
11//     fail to find child as parent deletion will trigger a cascading deletion of children(we
12//     could choose to skip child delete if we are sure original parent will be deleted but
13//     this is not easy to tell from deltaBom)
14//  3. for re-parent, child has to be updated to point to new parent before we
15//     delete old parent as otherwise cascade delete will kick in.
16// Logic: 
17//    first pass, we will move parent before child regardless of action code
18//    second pass, we will go from last to first and move lines with action=delete to the end of array
19 
20abocontext = util.abo_getContext();
21bDiagnosisOff=jsonget(abocontext, "AboDiagnosticDisabled", "boolean", true);  
22callContext="sortDeltaBom";
23 
24//constant declaration
25ACTION = "oRCL_ABO_ActionCode_l";
26DELETE =  "DELETE";
27 
28 
29//Step1: we will loop through the childarray 20 times, and move them to sortedChildJsonarray in the order of hierarchy
30 
31//mapId2Id store the list of item has been moved from original child to sortedChildJsonarray
32mapId2Id =  dict("string");
33//populate root into mapId2Id.
34id =  jsonget(deltaBom, "id", "string");
35put(mapId2Id, id, id);
36 
37//Read children to prepare childAsset
38childBomJsonArr =  jsonget(deltaBom, "children", "jsonarray", jsonarray());
39sizeofChildren =  jsonarraysize(childBomJsonArr);
40sortedChildjsonarray = jsonarray();
41 
42 
43// Assuming maximum depth of BOM is 20.
44dummyarray2 = string[20 * sizeofChildren];
45index= 0;
46for item in dummyArray2{
47     if(sizeofChildren==0){
48        break;  
49     } 
50     
51      curNode = jsonarrayget(childBomJsonArr, index, "json");        
52      parId =  jsonget(curNode, "parentId", "string", "");
53     
54     if(parId<>"" AND containsKey(mapId2Id, parId)){ 
55        //parent has already been processed 
56        jsonarrayappend(sortedChildjsonarray, curNode);
57        id =  jsonget(curNode, "id", "string");
58        put(mapId2Id, id, id);  
59        //jsonarrayremove returns the new size of array after removing element at index
60        sizeofChildren = jsonarrayremove(childBomJsonArr, index);
61 
62     }else{ 
63        //only adjust index if we did not delete the item
64        //as for delete the index already points to next item
65        index =  index + 1;
66     }
67     
68     if(index == sizeofChildren){    
69         index =  0;
70     }  
71}
72 
73if (jsonarraysize(childBomJsonArr)>0){
74  errMsg="Failed to sort delta configuration BOM, either it is over 20 levels deep or has a dangling reference inside, please contact your system administrator for further diagnosis.";  
75  if(not bDiagnosisOff){
76     dummy=util.abo_addDiagnosticInfo(errMsg+" deltaBom's current state", deltaBom,"");
77     errMsg=errMsg+" childBomJsonArr="+jsonarraytostr(childBomJsonArr);
78  }
79  throwError(errMsg);
80}
81 
82//Step2: sort from end to start and move lines with 'DELETE' action code at the end
83sizeofChildren = jsonarraysize(sortedChildjsonarray);
84index = sizeofChildren;
85dummyArray =  string[sizeofChildren];
86for item in dummyArray{
87  index  =  index - 1;
88  curNode =  jsonarrayget(sortedChildjsonarray,index, "json");
89  fieldJson =  jsonget(curNode, "fields", "json", json());
90  actionCode  =  jsonget(fieldJson, ACTION, "string", "");
91  if(actionCode == DELETE){
92     jsonarrayremove(sortedChildjsonarray, index);
93     //add node to the end of the list
94     jsonarrayappend(sortedChildjsonarray, curNode);
95     
96  }
97 } 
98jsonput(deltaBom, "children", sortedChildjsonarray);
99return deltaBom;
