Function Name : Support Parts Hiding
Name : Nandhini Vurimella
Description : For hiding support parts
*****************************************************************/
productType="";

res = bmql("select Product_Type from BRCM_Part_Master where MPN = $selectedProductNumberInt_avg");
for r in res
{
productType = get(r,"Product_Type");
}

supportRequired="";
mainatinenceRequired="";
resSupport = bmql("select Support_Required,Maintenance_Required from BRCM_Part_Master_EXT where MPN = $selectedProductNumberInt_avg");
for re in resSupport
{
supportRequired = get(re,"Support_Required");
mainatinenceRequired= get(re,"Maintenance_Required");
}


 if(supportRequired == "Y" or mainatinenceRequired == "Y")
 {
 return false;
 }

return true;
